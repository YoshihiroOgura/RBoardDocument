  leds_write(1)
  sleep(1)
  leds_write(0)
  sleep(1)
end

while true

  sleep 1

  sleep 1
end

7-Segment LED

Using 74HC4511, you can easily output with leds_write(0,1,5,6pin). 

j = 0
while true
  leds_write(j)
  j = j + 1
  if(j==10)
    j = 0
  end
  sleep(1)
end
