Simple MML Playback


TBL_FREQ = [0x95A, 0x84D, 0x76A, 0x6EF, 0x639, 0x58C, 0x4F3,
            0x4AB, 0x428, 0x3B4, 0x37F, 0x31D, 0x2BF, 0x278]
TBL_LENGTH = {1=>2000, 2=>1000, 4=>500, 8=>250}

PWM.new()
PWM.pin(16)
PWM.start(2)

def tone( freq )
  if freq == 0
    PWM.cycle(0,4)
    return
  end

  PWM.cycle(freq ,4)
end


def play( mml )
  mnote = nil
  mlength = 4
  octabe = 0

  mml.each_char {|ch|
    # Musical scale instruction?
    if "CDEFGABR".index(ch)
      play_sub( mnote, octabe, mlength )
      mnote = ch
      next
    end

    # Note length?
    if TBL_LENGTH[ch.to_i]
      mlength = ch.to_i
      next
    end

    # Octave up/down?
    if ch == "<" || ch == ">"
      play_sub( mnote, octabe, mlength )
      if ch == "<"
        octabe += 1
      else
        octabe -= 1
      end
      mnote = nil
      next
    end
  }
  play_sub( mnote, octabe, mlength )
end


def play_sub( mnote, octabe, mlength )
  if !mnote
    return
  end
  if mnote == "R"
    sleep_ms( TBL_LENGTH[mlength] || 1 )
    return
  end

  idx = "CDEFGAB".index(mnote)
  if idx
    idx = octabe * 7 + idx
    if TBL_FREQ[idx]
      tone( TBL_FREQ[idx] )
      #display( idx )
      sleep_ms( TBL_LENGTH[mlength] || 1 )
      tone( 0 )
    end
    return
  end

end
